# Fixed Axis Camera Controller

Use WASD to move the box. This example is fixed on the Z axis.

import useBrowserOnly from '@hooks/useBrowserOnly';
import { TGame } from '@tedengine/ted';

export const Game = () => {
  const onBrowser = useBrowserOnly();
  if (!onBrowser) {
    return null;
  }
  return (
    <TGame
      game={new Worker(new URL('@examples/cameras/fixed-axis-controller.ts', import.meta.url))}
    />
  );
};

<Game />
