# Mouse

Mouse location is logged in console.

import useBrowserOnly from '@hooks/useBrowserOnly';
import { TGame } from '@tedengine/ted';

export const Game = () => {
  const onBrowser = useBrowserOnly();
  if (!onBrowser) {
    return null;
  }
  return (
    <TGame
      game={new Worker(new URL('@examples/input/mouse.ts', import.meta.url))}
    />
  );
};

<Game />
