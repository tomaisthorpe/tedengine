# Sound Example

import useBrowserOnly from '@hooks/useBrowserOnly';
import { TGame } from '@tedengine/ted';

export const Game = () => {
  const onBrowser = useBrowserOnly();
  if (!onBrowser) {
    return null;
  }
  return (
    <TGame
      game={
        new Worker(new URL('@examples/audio/sound-example.ts', import.meta.url))
      }
    />
  );
};

<Game />
