# Game Config

Simple example to show what can be configured.
This example shows 100% game width.

import useBrowserOnly from '@hooks/useBrowserOnly';
import { TGame } from '@tedengine/ted';

export const Game = () => {
  const onBrowser = useBrowserOnly();
  if (!onBrowser) {
    return null;
  }
  return (
    <TGame
      width="100%"
      height="600px"
      game={new Worker(new URL('@examples/game-config.ts', import.meta.url))}
    />
  );
};

<Game />
