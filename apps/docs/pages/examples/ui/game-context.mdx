# Game Context

This example shows how to use the game context to pass data between your UI and the game.

Clicking on 'Press this' will trigger an event which is being listened for by the game and it will log when it is received.

import useBrowserOnly from '@hooks/useBrowserOnly';
import Game from '@examples/ui/game-context';

export const GameExample = () => {
  const onBrowser = useBrowserOnly();
  if (!onBrowser) {
    return null;
  }
  return (
    <Game
      worker={
        new Worker(
          new URL('@examples/ui/game-context-game.ts', import.meta.url)
        )
      }
    />
  );
};

<GameExample />
