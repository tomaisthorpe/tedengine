# Rotating System

This example shows how to create a system that rotates an object.

import useBrowserOnly from '@docusaurus/useIsBrowser';
import { TGame } from '@tedengine/ted';

export const Game = () => {
  const onBrowser = useBrowserOnly();
  if (!onBrowser) {
    return null;
  }
  return (
    <TGame
      game={
        new Worker(
          new URL('@examples/ecs/rotating-system.ts', import.meta.url),
        )
      }
    />
  );
};

<Game />
