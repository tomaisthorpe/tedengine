# Many Colliders

This example demonstrates the performance of the engine when using many colliders.

:::info
Some recent changes in the engine have resulted in a performance regression when creating a large number of meshes.

This is a known issue and will be fixed in a future release.
:::


import useBrowserOnly from '@docusaurus/useIsBrowser';
import { TGame } from '@tedengine/ted';

export const Game = () => {
  const onBrowser = useBrowserOnly();
  if (!onBrowser) {
    return null;
  }
  return (
    <TGame
      game={
        new Worker(
          new URL('@examples/benchmarks/many-colliders.ts', import.meta.url),
        )
      }
    />
  );
};

<Game />
