# Top Down Controller

Extended simple controller with angle calculation for pointing towards mouse.

import useBrowserOnly from '@docusaurus/useIsBrowser';
import { TGame } from '@tedengine/ted';

export const Game = () => {
  const onBrowser = useBrowserOnly();
  if (!onBrowser) {
    return null;
  }
  return (
    <TGame
      game={
        new Worker(
          new URL('@examples/input/top-down-controller.ts', import.meta.url),
        )
      }
    />
  );
};

<Game />
