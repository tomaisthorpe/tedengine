# Box Mesh

The engine provides a `createBoxMesh` function to create a box mesh.

This example shows how to create a box mesh and add it to the world.

import useIsBrowser from '@docusaurus/useIsBrowser';
import { TGame } from '@tedengine/ted';

export function Game() {
  const onBrowser = useIsBrowser();
  if (!onBrowser) {
    return null;
  }
  return (
    <TGame
      game={
        new Worker(new URL('@examples/3d/box-mesh.ts', import.meta.url))
      }
    />
  );
}

<Game />
