# Textured Mesh

Textured meshes can also be loaded from an OBJ and MTL file.

This example shows how to load a textured mesh and add it to the world.

import useIsBrowser from '@docusaurus/useIsBrowser';
import { TGame } from '@tedengine/ted';

export const Game = () => {
  const onBrowser = useIsBrowser();
  if (!onBrowser) {
    return null;
  }
  return (
    <TGame
      aspectRatio="auto"
      game={
        new Worker(new URL('@examples/3d/textured-mesh.ts', import.meta.url))
      }
    />
  );
};

<Game />
