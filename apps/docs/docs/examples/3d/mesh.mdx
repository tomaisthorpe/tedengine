# Mesh

This example shows loading a mesh from an OBJ and MTL file.

The mesh is then added to the world and rendered.

import useIsBrowser from '@docusaurus/useIsBrowser';
import { TGame } from '@tedengine/ted';

export const Game = () => {
  const onBrowser = useIsBrowser();
  if (!onBrowser) {
    return null;
  }
  return (
    <TGame
      aspectRatio="auto"
      game={
        new Worker(new URL('@examples/3d/mesh.ts', import.meta.url))
      }
    />
  );
};

<Game />
