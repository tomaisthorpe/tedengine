# Particle System

This example shows a simple particle system. It uses instanced sprites to render the particles.

import useBrowserOnly from '@docusaurus/useIsBrowser';
import { TGame } from '@tedengine/ted';

export const Game = () => {
  const onBrowser = useBrowserOnly();
  if (!onBrowser) {
    return null;
  }
  return (
    <TGame
      game={
        new Worker(new URL('@examples/3d/particle-system.ts', import.meta.url))
      }
    />
  );
};

<Game />
