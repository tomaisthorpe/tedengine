# Tilesets

Tilesets are a way to create complex 2D environments quickly and easily.

The engine supports loading tilesets from [LDtk](https://ldtk.io/) exported JSON files.

import useBrowserOnly from '@docusaurus/useIsBrowser';
import { TGame } from '@tedengine/ted';

export const Game = () => {
  const onBrowser = useBrowserOnly();
  if (!onBrowser) {
    return null;
  }
  return (
    <TGame
      game={new Worker(new URL('@examples/2d/tilesets.ts', import.meta.url))}
    />
  );
};

<Game />
