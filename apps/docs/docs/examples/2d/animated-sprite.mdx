# Animated Sprites

Sprites can be animated by providing a sprite sheet and a frame rate.

import useBrowserOnly from '@docusaurus/useIsBrowser';
import { TGame } from '@tedengine/ted';

export const Game = () => {
  const onBrowser = useBrowserOnly();
  if (!onBrowser) {
    return null;
  }
  return (
    <TGame
      config={{
        renderWidth: 800,
        renderHeight: 600,
        imageRendering: 'pixelated',
      }}
      game={
        new Worker(
          new URL('@examples/2d/animated-sprite.ts', import.meta.url),
        )
      }
    />
  );
};

<Game />
