# Sprites

Sprites are the most basic 2D object in the engine.
They can be created from a texture or a [canvas](./sprite-canvas.mdx).

import useBrowserOnly from '@docusaurus/useIsBrowser';
import { TGame } from '@tedengine/ted';

export const Game = () => {
  const onBrowser = useBrowserOnly();
  if (!onBrowser) {
    return null;
  }
  return (
    <TGame
      game={
        new Worker(new URL('@examples/2d/sprite.ts', import.meta.url))
      }
    />
  );
};

<Game />
