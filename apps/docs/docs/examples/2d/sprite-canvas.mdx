# Sprite Canvas

import useBrowserOnly from '@docusaurus/useIsBrowser';
import { TGame } from '@tedengine/ted';

export const Game = () => {
  const onBrowser = useBrowserOnly();
  if (!onBrowser) {
    return null;
  }
  return (
    <TGame
      game={
        new Worker(new URL('@examples/2d/sprite-canvas.ts', import.meta.url))
      }
    />
  );
};

<Game />
