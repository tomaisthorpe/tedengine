# Orbit Camera

The orbit camera moves the camera around a target object.

You can manually rotate around the object by clicking and dragging.
This behaviour can be toggled when creating the orbit camera component.

import useBrowserOnly from '@docusaurus/useIsBrowser';
import { TGame } from '@tedengine/ted';

export const Game = () => {
  const onBrowser = useBrowserOnly();
  if (!onBrowser) {
    return null;
  }
  return (
    <TGame
      game={
        new Worker(
          new URL('@examples/cameras/orbit-camera.ts', import.meta.url),
        )
      }
    />
  );
};

<Game />
