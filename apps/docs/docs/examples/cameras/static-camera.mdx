# Static Camera

This example demonstrates a static camera looking at a rotating 3D object.
You can switch between perspective and orthographic cameras using the dropdown.

import useBrowserOnly from '@docusaurus/useIsBrowser';
import { TGame } from '@tedengine/ted';

export const Game = () => {
  const onBrowser = useBrowserOnly();
  if (!onBrowser) {
    return null;
  }
  return (
    <TGame
      game={
        new Worker(
          new URL('@examples/cameras/static-camera.ts', import.meta.url),
        )
      }
    />
  );
};

<Game />
