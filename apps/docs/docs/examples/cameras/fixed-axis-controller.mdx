# Fixed Axis Camera Controller

Use WASD to move the box. You can select which axis to fix using the select box.

import useBrowserOnly from '@docusaurus/useIsBrowser';
import { TGame } from '@tedengine/ted';

export const Game = () => {
  const onBrowser = useBrowserOnly();
  if (!onBrowser) {
    return null;
  }
  return (
    <TGame
      game={
        new Worker(
          new URL(
            '@examples/cameras/fixed-axis-controller.ts',
            import.meta.url,
          ),
        )
      }
    />
  );
};

<Game />
