# Follow Component Camera Controller

Use WASD to move the box, the camera will rotate towards it.

import useBrowserOnly from '@docusaurus/useIsBrowser';
import { TGame } from '@tedengine/ted';

export const Game = () => {
  const onBrowser = useBrowserOnly();
  if (!onBrowser) {
    return null;
  }
  return (
    <TGame
      game={
        new Worker(
          new URL(
            '@examples/cameras/follow-component-controller.ts',
            import.meta.url,
          ),
        )
      }
    />
  );
};

<Game />
