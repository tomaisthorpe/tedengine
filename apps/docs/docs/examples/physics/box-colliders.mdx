# Box & Sphere Colliders

This example shows how to create a box, sphere, and plane collider.

import useBrowserOnly from '@docusaurus/useIsBrowser';
import { TGame } from '@tedengine/ted';

export const Game = () => {
  const onBrowser = useBrowserOnly();
  if (!onBrowser) {
    return null;
  }
  return (
    <TGame
      game={
        new Worker(
          new URL('@examples/physics/box-colliders.ts', import.meta.url),
        )
      }
    />
  );
};

<Game />
