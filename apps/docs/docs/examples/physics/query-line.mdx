# Query Line

This example shows how to raycast a line in the world and get the hits.
The example will log the hits to the console.

import useBrowserOnly from '@docusaurus/useIsBrowser';
import { TGame } from '@tedengine/ted';

export const Game = () => {
  const onBrowser = useBrowserOnly();
  if (!onBrowser) {
    return null;
  }
  return (
    <TGame
      game={
        new Worker(new URL('@examples/physics/query-line.ts', import.meta.url))
      }
    />
  );
};

<Game />
