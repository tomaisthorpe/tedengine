# Forces

This example shows how to apply forces to a rigid body.

Use WASD to move the box.

import useBrowserOnly from '@docusaurus/useIsBrowser';
import { TGame } from '@tedengine/ted';

export const Game = () => {
  const onBrowser = useBrowserOnly();
  if (!onBrowser) {
    return null;
  }
  return (
    <TGame
      game={new Worker(new URL('@examples/physics/forces.ts', import.meta.url))}
    />
  );
};

<Game />
